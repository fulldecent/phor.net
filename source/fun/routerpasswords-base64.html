      curl http://www.phenoelit-us.org/dpl/dpl.html > EDITME calic EDITME # keep un/pass columns, save to tmp sort -u
      tmp | grep . | sed -e 's/(none)//g' | sed -e 's|n/a||g' | sed -e 's/\t/:/g' > tmp2 perl -e 'use
      MIME::Base64;while (<>){chomp;chomp($e=encode_base64($_));print
      "<tr><td>$e<td>$_\n"}' < tmp2 > tmp3
